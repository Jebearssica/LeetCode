#include <vector>
class Solution
{
public:
    int removeElement(std::vector<int> &nums, int val)
    {
        int left = 0;
        int right = nums.size();
        //直接把末尾元素拿来覆盖, 还能不移指针再次判断直到值不等
        while (left < right)
            (nums[left] == val) ? (nums[left] = nums[--right]) : left++;
        return right;
    }
};
