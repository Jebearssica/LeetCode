```c++
#include <vector>
#include <string>
#include <unordered_map>
using namespace std;
class Solution
{
public:
    vector<string> findRepeatedDnaSequences(string s)
    {
        vector<string> res;
        unordered_map<string, int> map;
        int length = s.size();
        for (int idx = 0; idx <= length - 10; idx++)
        {
            string substr = s.substr(idx, 10);
            if (++map[substr] == 2)
                res.push_back(substr);
        }
        return res;
    }
};
```
